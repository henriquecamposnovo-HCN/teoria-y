pub fn op_tensivo(a: i8, b: i8) -> i8 {
    // T.op: opostos → T.NEU · neutro transmite
    if a == b  { return a; }
    if a == -b { return 0; }
    if a == 0  { return b; }
    if b == 0  { return a; }
    0
}
